498  template <size_t N>
499  const Matrix cat(const Matrix * const (&a)[N]) throw (std::invalid_argument)
500  {
501  size_t row_total = a[0]->row;
502  size_t column_total = 0;
503 
504  typedef const Matrix* const item_type;
505  const item_type * const end = a + N;
506 
507  for (const item_type * it = a; it != end; ++it) {
508  if ((*it)->row != row_total) {
509  throw std::invalid_argument("串联的矩阵的行数不一致");
510  }
511  column_total += (*it)->column;
512  }
513 
514  Matrix result(row_total, column_total, Matrix::uninit_tag);
515  size_t column_covered = 0;
516  for (const item_type * it = a; it != end; ++it) { //数组循环
517  for (size_t i = 0; i < row_total; ++i) { //行循环
518  std::uninitialized_copy((*it)->p[i], (*it)->p[i] + (*it)->column,
519  result.p[i] + column_covered);
520  }
521  column_covered += (*it)->column;
522  }
523  return result;
524  }
