48  {
49  if (row != 0 && column != 0) {
50  p = (Type**) malloc(row * sizeof(Type*));
51  if (p == NULL) {
52  throw std::bad_alloc();
53  }
54  for (size_t i = 0; i != row; ++i) {
55  p[i] = (Type*) malloc(column * sizeof(Type));
56  if (p[i] == NULL) {
57  while (i != 0) {
58  --i;
59  free (p[i]);
60  }
61  free (p);
62  throw std::bad_alloc();
63  }
64  }
65  } else {
66  row = 0;
67  column = 0;
68  p = NULL;
69  }
