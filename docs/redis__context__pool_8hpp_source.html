82  N(N)
83  {
84  for (size_t i = 0; i < N; ++i) {
85  pool[i] = getContext(ip, port, ms);
86  if (!pool[i]) {

93  ContextPoolReference apply(pid_t pid, const std::chrono::milliseconds & retry_interval = std::chrono::milliseconds(200),
94  unsigned short try_times = 5)
95  {
96  while (try_times--) {
97 
98  for (size_t i = 0; i < N; ++i) {
99  if (pids[i] == 0) {
100  pids[i] = pid;
101  return ContextPoolReference(i, *this);
102  }
103  }
104  std::this_thread::sleep_for(retry_interval);
105  }
106  throw ContextPoolApplyFailedException();
107  }
