397  template <size_t N>
398  const Matrix cat(const Matrix (&a)[N]) throw (std::invalid_argument)
399  {
400  int row_total = a[0].row;
401  int column_total = a[0].column;
402  for (size_t i = 1; i < N; i++) {
403  if (a[i].row != row_total) {
404  throw std::invalid_argument("串联的矩阵的行数不一致");
405  } else {
406  column_total += a[i].column;
407  }
408  }
409 
410  Matrix result(row_total, column_total);
411  int column_covered = 0;
412  for (const Matrix *it = a, *end = a + N; it != end; ++it) { //数组循环
413  for (int j = 0; j < row_total; j++) { //行循环
414  std::copy(it->p[j], it->p[j] + it->column, result.p[j] + column_covered);
415  }
416  column_covered += it->column;
417  }
418  return result;
419  }
