393  template <size_t N>
394  const Matrix cat(const Matrix (&a)[N]) throw (std::invalid_argument)
395  {
396  int row_total = a[0].row;
397  int column_total = a[0].column;
398  for (size_t i = 1; i < N; i++) {
399  if (a[i].row != row_total) {
400  throw std::invalid_argument("串联的矩阵的行数不一致");
401  } else {
402  column_total += a[i].column;
403  }
404  }
405 
406  Matrix result(row_total, column_total);
407  int column_covered = 0;
408  for (const Matrix *it = a, *end = a + N; it != end; ++it) { //数组循环
409  for (int j = 0; j < row_total; j++) { //行循环
410  std::copy(it->p[j], it->p[j] + it->column, result.p[j] + column_covered);
411  }
412  column_covered += it->column;
413  }
414  return result;
415  }
